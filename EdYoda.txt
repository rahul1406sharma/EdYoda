// Question 2:
// Merge the two arrays as keys and values into an object.
// If the key names contain '-' then convert it to camelcasing. 
// If there are duplicate values in keys array then add one key but merge values into an array.


Answer:

     function mergeArrays(keys, values) {
        let CamelCaseKeyArray = []; let CamelCaseValueArray = [];
        for (let i = 0; i < keys.length; i++) {
            CamelCaseKeyArray.push(keys[i].replace(/-([a-z])/g, function (m, w) {
                return w.toUpperCase();

            })
            )
        }
        console.log(CamelCaseKeyArray);
        for (let i = 0; i < values.length; i++) {
            CamelCaseValueArray.push(values[i].replace(/-([a-z])/g, function (i, x) {
                return x.toUpperCase();

            })
            )
        }
        console.log(CamelCaseValueArray);
        let CamelCaseArray = CamelCaseValueArray.reduce(function (result, field, index) {
            result[CamelCaseKeyArray[index]] = field;
            return result;
        }, {})
       
        console.log(CamelCaseArray);
    }
